@abstractmethod
def close_button(self):
    '''Button to close the current screen or modal.'''

@abstractmethod
def app_logo(self):
    '''The logo of the redBus application.'''

@abstractmethod
def mobile_number_label(self):
    '''Label indicating the mobile number field.'''

@abstractmethod
def mobile_number_text(self):
    '''The mobile number entered by the user.'''

@abstractmethod
def edit_mobile_number_link(self):
    '''Link to edit the mobile number.'''

@abstractmethod
def one_time_password_label(self):
    '''Label indicating the One Time Password input field.'''

@abstractmethod
def otp_input_field_1(self):
    '''Input field for the first digit of the One Time Password.'''

@abstractmethod
def otp_input_field_2(self):
    '''Input field for the second digit of the One Time Password.'''

@abstractmethod
def otp_input_field_3(self):
    '''Input field for the third digit of the One Time Password.'''

@abstractmethod
def otp_input_field_4(self):
    '''Input field for the fourth digit of the One Time Password.'''

@abstractmethod
def otp_input_field_5(self):
    '''Input field for the fifth digit of the One Time Password.'''

@abstractmethod
def otp_input_field_6(self):
    '''Input field for the sixth digit of the One Time Password.'''

@abstractmethod
def verify_otp_button(self):
    '''Button to verify the entered One Time Password.'''

@abstractmethod
def resend_otp_button(self):
    '''Button to request a new One Time Password, showing a countdown.'''
