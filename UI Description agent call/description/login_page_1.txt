# Page type: login_page

@abstractmethod
def close_button(self):
    '''Button to close the current page or modal.'''

@abstractmethod
def redbus_logo(self):
    '''The logo of the redBus service.'''

@abstractmethod
def welcome_back_label(self):
    '''Text label welcoming the user back.'''

@abstractmethod
def verify_details_label(self):
    '''Text label instructing the user to verify their details.'''

@abstractmethod
def country_code_dropdown(self):
    '''Dropdown or selector for choosing the country code for the mobile number.'''

@abstractmethod
def mobile_number_field(self):
    '''Input field for the user to enter their mobile number.'''

@abstractmethod
def registration_error_message(self):
    '''Text label indicating that the entered number is not registered.'''

@abstractmethod
def generate_otp_button(self):
    '''Button to generate and send a One Time Password to the entered mobile number.'''

@abstractmethod
def or_separator(self):
    '''Text label separating the two login/verification options.'''

@abstractmethod
def continue_with_google_button(self):
    '''Button to continue or log in using a Google account.'''
