# Page type: ticket_details_page

```python
@abstractmethod
def back_arrow_icon(self):
    '''Icon to navigate back to the previous screen.'''

@abstractmethod
def page_title(self):
    '''Title of the current page, indicating it shows ticket details.'''

@abstractmethod
def departure_location_icon(self):
    '''Icon representing the departure location.'''

@abstractmethod
def departure_location_text(self):
    '''Text displaying the departure location.'''

@abstractmethod
def arrival_location_icon(self):
    '''Icon representing the arrival location.'''

@abstractmethod
def arrival_location_text(self):
    '''Text displaying the arrival location.'''

@abstractmethod
def time_range_text(self):
    '''Text displaying the departure and arrival times.'''

@abstractmethod
def large_qr_code_image(self):
    '''A large QR code image representing the ticket.'''

@abstractmethod
def small_qr_code_image(self):
    '''A smaller QR code image, possibly for a different purpose or scan point.'''

@abstractmethod
def scan_qr_label(self):
    '''Label instructing the user where to scan the smaller QR code.'''
```
