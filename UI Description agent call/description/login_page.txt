# Page type: login_page

@abstractmethod
def close_button(self):
    '''Button to close the current screen or modal.'''

@abstractmethod
def app_logo(self):
    '''The logo of the redBus application.'''

@abstractmethod
def welcome_back_text(self):
    '''Heading text welcoming the user back.'''

@abstractmethod
def verify_details_text(self):
    '''Instruction text prompting the user to verify their details.'''

@abstractmethod
def country_code_dropdown(self):
    '''Dropdown to select the country code for the mobile number.'''

@abstractmethod
def mobile_number_input_field(self):
    '''Input field for the user to enter their mobile number.'''

@abstractmethod
def generate_otp_button(self):
    '''Button to generate and send a One Time Password to the entered mobile number.'''

@abstractmethod
def or_separator_text(self):
    '''Text separating the two login/verification options.'''

@abstractmethod
def continue_with_google_button(self):
    '''Button to continue or log in using a Google account.'''
